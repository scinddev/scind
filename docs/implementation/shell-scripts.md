# Contrail Shell Integration Scripts

**Version**: 0.1.2-draft
**Date**: December 2024
**Status**: Design Phase

<!-- Migrated from specs/contrail-shell-integration.md -->

This document contains the complete shell integration scripts for Contrail. These scripts are embedded in the Contrail binary and output by `contrail init-shell {bash|zsh|fish}`.

For the specification of how shell integration works (architecture, completion delegation, etc.), see [Shell Integration Specification](../specs/shell-integration.md).

---

## Installation

```bash
# Bash
contrail init-shell bash >> ~/.bashrc

# Zsh
contrail init-shell zsh >> ~/.zshrc

# Fish
contrail init-shell fish >> ~/.config/fish/conf.d/contrail.fish
```

## Bash Script

```bash
# =============================================================================
# Contrail Shell Integration for Bash
# Generated by: contrail init-shell bash
# =============================================================================

# -----------------------------------------------------------------------------
# Standard Contrail CLI Completion
# -----------------------------------------------------------------------------
# Generated by Cobra's built-in completion.
# Uncomment the following line once the contrail binary is built:
# eval "$(contrail completion bash)"

# -----------------------------------------------------------------------------
# contrail-compose: Docker Compose Passthrough with Context Awareness
# -----------------------------------------------------------------------------

contrail-compose() {
    local prefix
    local -a prefix_args=()
    local -a passthrough_args=()
    local parsing_contrail_flags=true

    # Parse arguments: extract -w/-a flags for contrail, pass rest through
    while [[ $# -gt 0 ]]; do
        if $parsing_contrail_flags; then
            case "$1" in
                -w|--workspace)
                    prefix_args+=("$1" "$2")
                    shift 2
                    ;;
                --workspace=*)
                    prefix_args+=("$1")
                    shift
                    ;;
                -a|--app)
                    prefix_args+=("$1" "$2")
                    shift 2
                    ;;
                --app=*)
                    prefix_args+=("$1")
                    shift
                    ;;
                -*)
                    # Unknown flag - might be for docker compose
                    parsing_contrail_flags=false
                    passthrough_args+=("$1")
                    shift
                    ;;
                *)
                    # Positional argument - stop parsing contrail flags
                    parsing_contrail_flags=false
                    passthrough_args+=("$1")
                    shift
                    ;;
            esac
        else
            passthrough_args+=("$1")
            shift
        fi
    done

    # Get the docker compose prefix from contrail
    prefix=$(contrail compose-prefix "${prefix_args[@]}" 2>/dev/null)
    if [[ -z "$prefix" ]]; then
        # Run again to show the error message
        contrail compose-prefix "${prefix_args[@]}"
        return 1
    fi

    # Execute: eval handles the quoted paths in prefix correctly
    eval "$prefix" '"${passthrough_args[@]}"'
}

# -----------------------------------------------------------------------------
# Completion for contrail-compose
# Delegates to Docker's completion with transformed context
# -----------------------------------------------------------------------------

_contrail_compose_completions() {
    local cur prev words cword
    _init_completion || return

    # Extract contrail-specific flags to pass to compose-prefix
    local -a prefix_cmd_args=()
    local -a docker_words=()
    local parsing_contrail_flags=true
    local i=1  # Skip words[0] which is "contrail-compose"

    while [[ $i -lt ${#words[@]} ]]; do
        if $parsing_contrail_flags; then
            case "${words[$i]}" in
                -w|--workspace)
                    prefix_cmd_args+=("${words[$i]}" "${words[$((i+1))]}")
                    ((i += 2))
                    ;;
                --workspace=*)
                    prefix_cmd_args+=("${words[$i]}")
                    ((i++))
                    ;;
                -a|--app)
                    prefix_cmd_args+=("${words[$i]}" "${words[$((i+1))]}")
                    ((i += 2))
                    ;;
                --app=*)
                    prefix_cmd_args+=("${words[$i]}")
                    ((i++))
                    ;;
                -*)
                    parsing_contrail_flags=false
                    docker_words+=("${words[$i]}")
                    ((i++))
                    ;;
                *)
                    parsing_contrail_flags=false
                    docker_words+=("${words[$i]}")
                    ((i++))
                    ;;
            esac
        else
            docker_words+=("${words[$i]}")
            ((i++))
        fi
    done

    # Get prefix from contrail
    local prefix
    prefix=$(contrail compose-prefix "${prefix_cmd_args[@]}" 2>/dev/null)
    [[ -z "$prefix" ]] && return

    # Parse prefix into an array (handles quoted paths)
    local -a prefix_words
    eval "prefix_words=($prefix)"

    # Build fake command line for docker's completion
    # Original: contrail-compose exec php
    # Fake:     docker compose -p X -f Y exec php
    local -a fake_words=("${prefix_words[@]}" "${docker_words[@]}")
    local fake_cword=$(( ${#prefix_words[@]} + ${#docker_words[@]} - 1 ))

    # Adjust fake_cword if we're completing a new word
    if [[ -z "$cur" ]] || [[ "$cur" != "${words[$cword]}" ]]; then
        ((fake_cword++))
    fi

    # Check if docker completion function exists
    if ! declare -F _docker &>/dev/null; then
        # Fallback: try to load docker completion
        if [[ -f /usr/share/bash-completion/completions/docker ]]; then
            source /usr/share/bash-completion/completions/docker
        elif [[ -f /etc/bash_completion.d/docker ]]; then
            source /etc/bash_completion.d/docker
        elif [[ -f "$(brew --prefix 2>/dev/null)/etc/bash_completion.d/docker" ]]; then
            source "$(brew --prefix)/etc/bash_completion.d/docker"
        fi
    fi

    # If still no docker completion, provide basic subcommand completion
    if ! declare -F _docker &>/dev/null; then
        if [[ ${#docker_words[@]} -eq 0 ]]; then
            COMPREPLY=($(compgen -W "build config create down events exec images kill logs pause port ps pull push restart rm run start stop top unpause up version" -- "$cur"))
        fi
        return
    fi

    # Save original completion context
    local orig_cword=$cword
    local -a orig_words=("${words[@]}")

    # Override with fake context
    words=("${fake_words[@]}")
    cword=$fake_cword
    COMP_WORDS=("${fake_words[@]}")
    COMP_CWORD=$fake_cword
    COMP_LINE="${fake_words[*]}"
    COMP_POINT=${#COMP_LINE}

    # Invoke docker's completion
    _docker

    # Restore original context (defensive)
    words=("${orig_words[@]}")
    cword=$orig_cword
}

# Register completion
complete -F _contrail_compose_completions contrail-compose

# -----------------------------------------------------------------------------
# Optional: Short alias (users can uncomment if desired)
# -----------------------------------------------------------------------------
# alias cc='contrail-compose'
```

```zsh
# -----------------------------------------------------------------------------
# Completion for contrail-compose
# Delegates to Docker's completion with transformed context
# -----------------------------------------------------------------------------

_contrail-compose() {
    local -a prefix_cmd_args=()
    local -a docker_words=()
    local parsing_contrail_flags=true
    local i=2  # words[1] is "contrail-compose" in zsh (1-indexed)

    # Extract contrail-specific flags, pass rest to docker
    while [[ $i -le ${#words[@]} ]]; do
        if $parsing_contrail_flags; then
            case "${words[$i]}" in
                -w|--workspace)
                    prefix_cmd_args+=("${words[$i]}" "${words[$((i+1))]}")
                    ((i += 2))
                    ;;
                --workspace=*)
                    prefix_cmd_args+=("${words[$i]}")
                    ((i++))
                    ;;
                -a|--app)
                    prefix_cmd_args+=("${words[$i]}" "${words[$((i+1))]}")
                    ((i += 2))
                    ;;
                --app=*)
                    prefix_cmd_args+=("${words[$i]}")
                    ((i++))
                    ;;
                -*)
                    parsing_contrail_flags=false
                    docker_words+=("${words[$i]}")
                    ((i++))
                    ;;
                *)
                    parsing_contrail_flags=false
                    docker_words+=("${words[$i]}")
                    ((i++))
                    ;;
            esac
        else
            docker_words+=("${words[$i]}")
            ((i++))
        fi
    done

    # Get prefix from contrail
    local prefix
    prefix=$(contrail compose-prefix "${prefix_cmd_args[@]}" 2>/dev/null)
    [[ -z "$prefix" ]] && return 1

    # Parse prefix into array (respecting quotes)
    # (z) splits like shell words, (@Q) removes quotes while preserving array
    local -a prefix_words
    prefix_words=("${(@Q)${(z)prefix}}")

    # Build fake command line for docker's completion
    # Zsh arrays are 1-indexed, CURRENT is 1-indexed position
    #
    # Original: contrail-compose exec php
    #   words=("contrail-compose" "exec" "php") CURRENT=3 (or 4 if completing new word)
    #
    # Fake: docker compose -p X -f Y exec php
    #   words=("docker" "compose" "-p" "X" "-f" "Y" "exec" "php")

    local -a fake_words=("${prefix_words[@]}" "${docker_words[@]}")

    # Calculate new CURRENT position
    # Original CURRENT is relative to words array
    # We need to map it to the fake_words array
    local words_before_docker=$((CURRENT - 1))  # How many words typed (excluding current)
    local contrail_flags_consumed=$((${#words[@]} - 1 - ${#docker_words[@]}))
    local fake_current=$((${#prefix_words[@]} + ${#docker_words[@]}))

    # If we're in the middle of typing a word, adjust
    if [[ $CURRENT -le ${#words[@]} ]] && [[ -n "${words[$CURRENT]}" ]]; then
        fake_current=$((${#prefix_words[@]} + CURRENT - 1 - contrail_flags_consumed))
    fi

    # Override completion context
    words=("${fake_words[@]}")
    CURRENT=$fake_current

    # Try to find docker's completion function
    # It might be _docker, _docker-compose, or loaded lazily
    if (( $+functions[_docker] )); then
        _docker
    elif (( $+functions[_docker-compose] )); then
        _docker-compose
    else
        # Fallback: basic subcommand completion
        if [[ ${#docker_words[@]} -eq 0 ]]; then
            local -a subcommands=(
                'build:Build or rebuild services'
                'config:Validate and view the Compose file'
                'create:Create services'
                'down:Stop and remove containers, networks'
                'events:Receive real time events from containers'
                'exec:Execute a command in a running container'
                'images:List images used by the created containers'
                'kill:Kill containers'
                'logs:View output from containers'
                'pause:Pause services'
                'port:Print the public port for a port binding'
                'ps:List containers'
                'pull:Pull service images'
                'push:Push service images'
                'restart:Restart services'
                'rm:Remove stopped containers'
                'run:Run a one-off command'
                'start:Start services'
                'stop:Stop services'
                'top:Display the running processes'
                'unpause:Unpause services'
                'up:Create and start containers'
                'version:Show version information'
            )
            _describe 'docker compose command' subcommands
        fi
    fi
}

# Register completion
compdef _contrail-compose contrail-compose

# -----------------------------------------------------------------------------
# Optional: Short alias (users can uncomment if desired)
# -----------------------------------------------------------------------------
# alias cc='contrail-compose'
```

## Fish Script

```fish
# =============================================================================
# Contrail Shell Integration for Fish
# Generated by: contrail init-shell fish
# =============================================================================

# -----------------------------------------------------------------------------
# Standard Contrail CLI Completion
# -----------------------------------------------------------------------------
# Generated by Cobra's built-in completion.
# Uncomment the following line once the contrail binary is built:
# contrail completion fish | source

# -----------------------------------------------------------------------------
# contrail-compose: Docker Compose Passthrough with Context Awareness
# -----------------------------------------------------------------------------

function contrail-compose --description "Docker Compose with Contrail context awareness"
    set -l prefix_args
    set -l passthrough_args
    set -l parsing_contrail_flags true

    # Parse arguments: extract -w/-a flags for contrail, pass rest through
    set -l i 1
    while test $i -le (count $argv)
        if test "$parsing_contrail_flags" = true
            switch $argv[$i]
                case -w --workspace
                    set -a prefix_args $argv[$i] $argv[(math $i + 1)]
                    set i (math $i + 2)
                case '--workspace=*'
                    set -a prefix_args $argv[$i]
                    set i (math $i + 1)
                case -a --app
                    set -a prefix_args $argv[$i] $argv[(math $i + 1)]
                    set i (math $i + 2)
                case '--app=*'
                    set -a prefix_args $argv[$i]
                    set i (math $i + 1)
                case '-*'
                    # Unknown flag - might be for docker compose
                    set parsing_contrail_flags false
                    set -a passthrough_args $argv[$i]
                    set i (math $i + 1)
                case '*'
                    # Positional argument - stop parsing contrail flags
                    set parsing_contrail_flags false
                    set -a passthrough_args $argv[$i]
                    set i (math $i + 1)
            end
        else
            set -a passthrough_args $argv[$i]
            set i (math $i + 1)
        end
    end

    # Get the docker compose prefix from contrail
    set -l prefix (contrail compose-prefix $prefix_args 2>/dev/null)
    if test -z "$prefix"
        # Run again to show the error message
        contrail compose-prefix $prefix_args
        return 1
    end

    # Execute: eval handles the quoted paths in prefix correctly
    eval $prefix $passthrough_args
end

# -----------------------------------------------------------------------------
# Completion for contrail-compose
# Uses fish's complete -C to delegate to docker's completion
# -----------------------------------------------------------------------------

function __contrail_compose_complete
    # Get the current command line tokens
    set -l tokens (commandline -opc)

    # Extract contrail-specific flags, build docker command
    set -l prefix_args
    set -l docker_args
    set -l parsing_contrail_flags true
    set -l i 2  # Skip "contrail-compose"

    while test $i -le (count $tokens)
        if test "$parsing_contrail_flags" = true
            switch $tokens[$i]
                case -w --workspace
                    set -a prefix_args $tokens[$i] $tokens[(math $i + 1)]
                    set i (math $i + 2)
                case '--workspace=*'
                    set -a prefix_args $tokens[$i]
                    set i (math $i + 1)
                case -a --app
                    set -a prefix_args $tokens[$i] $tokens[(math $i + 1)]
                    set i (math $i + 2)
                case '--app=*'
                    set -a prefix_args $tokens[$i]
                    set i (math $i + 1)
                case '-*'
                    set parsing_contrail_flags false
                    set -a docker_args $tokens[$i]
                    set i (math $i + 1)
                case '*'
                    set parsing_contrail_flags false
                    set -a docker_args $tokens[$i]
                    set i (math $i + 1)
            end
        else
            set -a docker_args $tokens[$i]
            set i (math $i + 1)
        end
    end

    # Get prefix from contrail
    set -l prefix (contrail compose-prefix $prefix_args 2>/dev/null | string split ' ')
    if test -z "$prefix"
        return
    end

    # Build fake command line and get completions
    # The trailing space is important - it tells fish we're completing the next word
    set -l fake_line (string join ' ' $prefix $docker_args)

    # Check if we're completing a partial word
    set -l current_token (commandline -ct)
    if test -n "$current_token"
        set fake_line "$fake_line$current_token"
    else
        set fake_line "$fake_line "
    end

    # Use fish's built-in completion delegation
    complete -C "$fake_line"
end

# Provide contrail-specific flag completions first
complete -c contrail-compose -s w -l workspace -d "Specify workspace" -xa "(contrail workspace list -q 2>/dev/null)"
complete -c contrail-compose -s a -l app -d "Specify application" -xa "(contrail app list -q 2>/dev/null)"

# Delegate everything else to docker compose completion
complete -c contrail-compose -a "(__contrail_compose_complete)"

# -----------------------------------------------------------------------------
# Optional: Short abbreviation (users can uncomment if desired)
# -----------------------------------------------------------------------------
# abbr -a cc contrail-compose
```

## Known Limitations

### All Shells

1. **Docker completion must be installed**: If docker's shell completion isn't available, the delegation falls back to basic subcommand completion only.

2. **First invocation latency**: The first tab completion calls `contrail compose-prefix`, which involves context detection. Subsequent completions in the same directory use shell caching.

3. **Path spaces**: Paths with spaces are supported but require careful quoting in `contrail compose-prefix` output.

### Bash-Specific

1. **Requires bash-completion package**: The `_init_completion` helper function comes from bash-completion. Without it, completion may not work.

2. **COMP_LINE reconstruction**: Some docker completions use `COMP_LINE` for substring matching. Our reconstruction from words may lose original spacing.

### Zsh-Specific

1. **1-indexed arrays**: The completion function must account for zsh's 1-indexed arrays when calculating positions.

2. **Completion function detection**: Docker's completion might be `_docker`, `_docker-compose`, or loaded lazily. The script tries multiple approaches.

### Fish-Specific

1. **`complete -C` overhead**: Fish's completion delegation spawns a subshell, adding latency compared to bash/zsh.

2. **Completion caching**: Fish aggressively caches completions. If compose files change, completions may be stale until the shell restarts.
